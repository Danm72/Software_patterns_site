1    package com.dan.toyapp.database; 
2     
3    import android.content.Context; 
4    import android.database.sqlite.SQLiteDatabase; 
5    import android.database.sqlite.SQLiteOpenHelper; 
6    import android.util.Log; 
7     
8    /** 
9     * Created by danmalone on 18/10/2013.

11   public class SqlLiteHelperCountries extends SQLiteOpenHelper { 
12    
13       public static final String TABLE_COUNTRIES = "countries"; 
14       public static final String COLUMN_CODE = "code"; 
15       public static final String COLUMN_NAME = "name"; 
16    
17       private static final String DATABASE_NAME = "countries.db"; 
18       private static final int DATABASE_VERSION = 1; 
19    
20       // Database creation sql statement 
21       private static final String DATABASE_CREATE = "create table " 
22               + TABLE_COUNTRIES + "(" + COLUMN_CODE 
23               + " text not null primary key,"+ COLUMN_NAME + " text not null)"; 
24    
25       public SqlLiteHelperCountries(Context context) { 
26           super(context, DATABASE_NAME, null, DATABASE_VERSION); 
27       } 
28    
29       @Override 
30       public void onCreate(SQLiteDatabase database) { 
31           database.execSQL(DATABASE_CREATE); 
32       } 
33    
34       @Override 
35       public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) { 
36           Log.w(SqlLiteHelperCountries.class.getName(), 
37                   "Upgrading database from version " + oldVersion + " to " 
38                           + newVersion + ", which will destroy all old data"); 
39           db.execSQL("DROP TABLE IF EXISTS " + TABLE_COUNTRIES); 
40           onCreate(db); 
41       } 
42    
43   }
